```typescript
import { requireCFI } from '@/lib/auth'

export async function GET(request: NextRequest) {
  const authResult = await requireCFI(request)
  if ('error' in authResult) return authResult.error
  const { user } = authResult
  const db = getSupabaseAdmin()

  // Get flight logs where instructor_id = this CFI
  const { data, error } = await db
    .from('student_lesson_completions')
    .select('id, student_id, instructor_id, completed_at, notes, students!inner(full_name)')
    .eq('instructor_id', user.id)
    .order('completed_at', { ascending: false })
    .limit(50)

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })
  return NextResponse.json(data || [])
}
```

**POST handler** — log flight hours (per D-13, D-14):
```typescript
export async function POST(request: NextRequest) {
  const authResult = await requireCFI(request)
  if ('error' in authResult) return authResult.error
  const { user } = authResult
  const db = getSupabaseAdmin()

  const body = await request.json()
  const { student_id, hours, date, notes } = body

  // Validate required fields (per D-14)
  if (!student_id || typeof student_id !== 'string') {
    return NextResponse.json({ error: 'student_id is required' }, { status: 400 })
  }
  if (!hours || typeof hours !== 'number' || hours <= 0 || hours > 24) {
    return NextResponse.json({ error: 'hours must be a number between 0.1 and 24' }, { status: 400 })
  }
  if (!date || typeof date !== 'string') {
    return NextResponse.json({ error: 'date is required' }, { status: 400 })
  }

  // SECURITY: Verify student is in this CFI's roster (per D-02, T-03-05)
  const { data: student, error: studentError } = await db
    .from('students')
    .select('id, user_id, full_name')
    .eq('user_id', student_id)
    .eq('instructor_id', user.id)
    .single()

  if (studentError || !student) {
    return NextResponse.json({ error: 'Student not found in your roster' }, { status: 403 })
  }

  // Step 1: Insert into student_lesson_completions (per D-13)
  const { error: insertError } = await db
    .from('student_lesson_completions')
    .insert({
      student_id: student_id,
      instructor_id: user.id,
      completed_at: date,
      notes: notes || null,
      // syllabus_lesson_id is now nullable (03-01 migration)
    })

  if (insertError) {
    return NextResponse.json({ error: insertError.message }, { status: 500 })
  }

  // Step 2: Atomic increment of student hours (per D-13, RESEARCH anti-pattern)
  // Use raw SQL via Supabase RPC or atomic update syntax
  // Supabase JS client does not support `column = column + value` directly.
  // Use .rpc() with a custom function, OR use two separate updates.
  // Safest approach without a custom RPC function:
  const { error: hoursError } = await db.rpc('increment_student_hours', {
    p_student_user_id: student_id,
    p_hours: hours,
  })

  // If RPC function doesn't exist, fall back to raw update:
  // This is NOT atomic but acceptable for v1 with single-CFI concurrency.
  // TODO: Create the RPC function in SETUP.sql if it doesn't exist.
  if (hoursError) {
    // Fallback: direct update (check SETUP.sql for existing RPC)
    const { error: updateError } = await db
      .from('students')
      .update({
        dual_hours: student.dual_hours + hours,  // Read current + add
        total_hours: student.total_hours + hours,
      })
      .eq('user_id', student_id)

    // Note: This fallback is not atomic. If the RPC doesn't exist,
    // create it (see below in SETUP.sql addition).
  }

  return NextResponse.json({ success: true, student_name: student.full_name, hours_logged: hours }, { status: 201 })
}
```

**IMPORTANT: Add RPC function to `supabase/SETUP.sql`** for atomic hour increment (per RESEARCH "Don't Hand-Roll" — use Postgres `SET dual_hours = dual_hours + $1`):
```sql
-- Phase 3: Atomic flight hour increment (avoids race condition per T-03-05)
CREATE OR REPLACE FUNCTION increment_student_hours(p_student_user_id UUID, p_hours NUMERIC)
RETURNS VOID AS $$
BEGIN
  UPDATE students
  SET dual_hours = COALESCE(dual_hours, 0) + p_hours,
      total_hours = COALESCE(total_hours, 0) + p_hours
  WHERE user_id = p_student_user_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```
Append this function definition to the end of `supabase/SETUP.sql`. `SECURITY DEFINER` allows the service role client to call it. The API route should use `db.rpc('increment_student_hours', { p_student_user_id, p_hours })` as the primary path, with the fallback update removed once the RPC is confirmed to exist.

The final POST handler should use ONLY the RPC call (not the fallback):
```typescript
const { error: hoursError } = await db.rpc('increment_student_hours', {
  p_student_user_id: student_id,
  p_hours: hours,
})
if (hoursError) {
  return NextResponse.json({ error: 'Failed to update student hours' }, { status: 500 })
}
```

```typescript
export async function POST(request: NextRequest) {
  const authResult = await requireCFI(request)
  if ('error' in authResult) return authResult.error
  const { user } = authResult
  const db = getSupabaseAdmin()

  const body = await request.json()
  const { student_id, endorsement_type, notes } = body

  // Validate required fields
  if (!student_id || typeof student_id !== 'string') {
    return NextResponse.json({ error: 'student_id is required' }, { status: 400 })
  }

  const validTypes = ['solo', 'xc_solo', 'night_solo', 'checkride_prep', 'instrument_proficiency_check', 'flight_review', 'other']
  if (!endorsement_type || !validTypes.includes(endorsement_type)) {
    return NextResponse.json({
      error: `endorsement_type must be one of: ${validTypes.join(', ')}`
    }, { status: 400 })
  }

  // SECURITY: Verify student is in this CFI's roster
  const { data: student, error: studentError } = await db
    .from('students')
    .select('id, user_id, full_name')
    .eq('user_id', student_id)
    .eq('instructor_id', user.id)
    .single()

  if (studentError || !student) {
    return NextResponse.json({ error: 'Student not found in your roster' }, { status: 403 })
  }

  // Insert endorsement (per D-15)
  const { data: endorsement, error: insertError } = await db
    .from('student_endorsements')
    .insert({
      student_id: student_id,
      instructor_id: user.id,
      endorsement_type: endorsement_type,
      notes: notes || null,
      // endorsed_at defaults to NOW() in Postgres
    })
    .select()
    .single()

  if (insertError) {
    return NextResponse.json({ error: insertError.message }, { status: 500 })
  }

  return NextResponse.json(endorsement, { status: 201 })
}
```

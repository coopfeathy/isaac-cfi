**Part A — Convert manage/layout.tsx to Server Component (SEC-01, SEC-02):**

Replace the entire file. Remove `'use client'`. Make it an async Server Component that:
1. Creates a server-side Supabase client using `createServerClient` from `@supabase/ssr` with `cookies()` from `next/headers`
2. Calls `getUser()` to verify session (middleware already redirects unauthenticated users, but defense-in-depth)
3. Queries `profiles.is_admin` for the authenticated user
4. If not admin, calls `redirect('/dashboard')` (SEC-02 — non-admin redirect)
5. Renders the existing layout structure with `ManageSidebar` and `{children}`

```typescript
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'
import { redirect } from 'next/navigation'
import ManageSidebar from '../components/ManageSidebar'

export default async function ManageLayout({
  children,
}: {
  children: React.ReactNode
}) {
  const cookieStore = await cookies()

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll()
        },
        setAll() {
          // Server Components cannot set cookies; middleware handles refresh
        },
      },
    }
  )

  const { data: { user } } = await supabase.auth.getUser()

  if (!user) {
    redirect('/login')
  }

  // Check admin role (SEC-02)
  const { data: profile } = await supabase
    .from('profiles')
    .select('is_admin')
    .eq('id', user.id)
    .single()

  if (!profile?.is_admin) {
    redirect('/dashboard')
  }

  return (
    <div style={{ display: 'flex', minHeight: '100vh', backgroundColor: '#f3f4f6' }}>
      <ManageSidebar />
      <main style={{
        marginLeft: '280px',
        flex: 1,
        padding: '32px',
        width: 'calc(100% - 280px)'
      }}>
        {children}
      </main>
    </div>
  )
}
```

Note: `cookies()` in Next.js 16 is `async` — use `await cookies()`. The `setAll` callback is a no-op in Server Components because they cannot set cookies; the middleware handles session refresh.

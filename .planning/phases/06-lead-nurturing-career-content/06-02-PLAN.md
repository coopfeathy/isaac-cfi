    2. Create `lib/ratelimit.ts`:
       ```typescript
       import { Ratelimit } from '@upstash/ratelimit'
       import { Redis } from '@upstash/redis'
       import { NextRequest, NextResponse } from 'next/server'

       let _ratelimit: Ratelimit | null = null

       function getRatelimit(): Ratelimit | null {
         if (!process.env.UPSTASH_REDIS_REST_URL || !process.env.UPSTASH_REDIS_REST_TOKEN) {
           console.warn('Upstash env vars missing — rate limiting disabled')
           return null
         }
         if (!_ratelimit) {
           _ratelimit = new Ratelimit({
             redis: Redis.fromEnv(),
             limiter: Ratelimit.slidingWindow(10, '1 h'),
             prefix: 'merlin_rl',
           })
         }
         return _ratelimit
       }

       export async function applyRateLimit(request: NextRequest): Promise<NextResponse | null> {
         const limiter = getRatelimit()
         if (!limiter) return null  // fail open if not configured

         const ip = request.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ?? '127.0.0.1'
         try {
           const { success } = await limiter.limit(ip)
           if (!success) {
             return NextResponse.json(
               { error: 'Too many requests — please try again later.' },
               { status: 429 }
             )
           }
         } catch (err) {
           console.error('Rate limit check failed:', err)
           // fail open — allow request if Redis is down
         }
         return null
       }

       export { getRatelimit }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> API routes | Untrusted form submissions from public internet; rate limiting is the control |
| API routes -> Upstash Redis | Server-side only; REST token in env vars |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-06 | Denial of Service | discovery-flight-signup, contact, slot-requests | mitigate | `@upstash/ratelimit` slidingWindow(10, '1 h') per IP applied at route entry before any DB/email work |
| T-06-07 | Information Disclosure | lib/ratelimit.ts | mitigate | Upstash credentials read from server-side env vars via `Redis.fromEnv()`; never imported in 'use client' components |
| T-06-08 | Denial of Service | lib/ratelimit.ts (Redis down) | mitigate | Fail-open pattern: if Redis unreachable or env vars missing, request passes through with console warning — prevents self-DoS |
| T-06-09 | Spoofing | x-forwarded-for header | accept | IP extracted from x-forwarded-for (standard for proxied environments like Netlify); spoofing this header can bypass rate limit but requires network-level access — acceptable risk for L1 |
</threat_model>
